192 def get_data_variation(self, arr1, arr2, aligned=True, contig=True):
193 """
194 Returns a copy of arr1 that may be non-contiguous or unaligned, and a
195 matching array for arr2 (although not a copy).
196 """
197 if contig:
198 stride1 = arr1.dtype.itemsize
199 stride2 = arr2.dtype.itemsize
200 elif aligned:
201 stride1 = 2 * arr1.dtype.itemsize
202 stride2 = 2 * arr2.dtype.itemsize
203 else:
204 stride1 = arr1.dtype.itemsize + 1
205 stride2 = arr2.dtype.itemsize + 1
206  
207 max_size1 = len(arr1) * 3 * arr1.dtype.itemsize + 1
208 max_size2 = len(arr2) * 3 * arr2.dtype.itemsize + 1
209 from_bytes = np.zeros(max_size1, dtype=np.uint8)
210 to_bytes = np.zeros(max_size2, dtype=np.uint8)
211  
212 # Sanity check that the above is large enough:
213 assert stride1 * len(arr1) <= from_bytes.nbytes
214 assert stride2 * len(arr2) <= to_bytes.nbytes
215  
216 if aligned:
217 new1 = as_strided(from_bytes[:-1].view(arr1.dtype),
218 arr1.shape, (stride1,))
219 new2 = as_strided(to_bytes[:-1].view(arr2.dtype),
220 arr2.shape, (stride2,))
221 else:
222 new1 = as_strided(from_bytes[1:].view(arr1.dtype),
223 arr1.shape, (stride1,))
224 new2 = as_strided(to_bytes[1:].view(arr2.dtype),
225 arr2.shape, (stride2,))
226  
227 new1[...] = arr1
228  
229 if not contig:
230 # Ensure we did not overwrite bytes that should not be written:
231 offset = arr1.dtype.itemsize if aligned else 0
232 buf = from_bytes[offset::stride1].tobytes()

235 if contig:
236 assert new1.flags.c_contiguous
237 assert new2.flags.c_contiguous
238 else:
239 assert not new1.flags.c_contiguous
240 assert not new2.flags.c_contiguous

244 assert new2.flags.aligned
