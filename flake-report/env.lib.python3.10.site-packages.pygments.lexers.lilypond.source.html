55 # Because parsing LilyPond input is very tricky (and in fact
56 # impossible without executing LilyPond when there is Scheme
57 # code in the file), this lexer does not try to recognize
58 # lexical modes. Instead, it catches the most frequent pieces
59 # of syntax, and, above all, knows about many kinds of builtins.
60  
61 # In order to parse embedded Scheme, this lexer subclasses the SchemeLexer.
62 # It redefines the 'root' state entirely, and adds a rule for #{ #}
63 # to the 'value' state. The latter is used to parse a Scheme expression
64 # after #.
65  
66 def get_tokens_unprocessed(self, text):
67 """Highlight Scheme variables as LilyPond builtins when applicable."""
68 for index, token, value in super().get_tokens_unprocessed(text):
69 if token is Token.Name.Function or token is Token.Name.Variable:
70 if value in scheme_functions:
71 token = Token.Name.Builtin.SchemeFunction
72 elif token is Token.Name.Builtin:
73 token = Token.Name.Builtin.SchemeBuiltin
74 yield index, token, value
