620 def _process_unlinkable_fobjects(self, objects, libraries,
621 fcompiler, library_dirs,
622 unlinkable_fobjects):
623 libraries = list(libraries)
624 objects = list(objects)
625 unlinkable_fobjects = list(unlinkable_fobjects)
626  
627 # Expand possible fake static libraries to objects;
628 # make sure to iterate over a copy of the list as
629 # "fake" libraries will be removed as they are
630 # encountered
631 for lib in libraries[:]:
632 for libdir in library_dirs:
633 fake_lib = os.path.join(libdir, lib + '.fobjects')
634 if os.path.isfile(fake_lib):
635 # Replace fake static library
636 libraries.remove(lib)
637 with open(fake_lib, 'r') as f:
638 unlinkable_fobjects.extend(f.read().splitlines())
639  
640 # Expand C objects
641 c_lib = os.path.join(libdir, lib + '.cobjects')
642 with open(c_lib, 'r') as f:
643 objects.extend(f.read().splitlines())
644  
645 # Wrap unlinkable objects to a linkable one
646 if unlinkable_fobjects:
647 fobjects = [os.path.abspath(obj) for obj in unlinkable_fobjects]
648 wrapped = fcompiler.wrap_unlinkable_objects(
649 fobjects, output_dir=self.build_temp,
650 extra_dll_dir=self.extra_dll_dir)
651 objects.extend(wrapped)
652  
653 return objects, libraries

655 def _libs_with_msvc_and_fortran(self, fcompiler, c_libraries,
656 c_library_dirs):
657 if fcompiler is None:
658 return
659  
660 for libname in c_libraries:
661 if libname.startswith('msvc'):
662 continue
663 fileexists = False
664 for libdir in c_library_dirs or []:
665 libfile = os.path.join(libdir, '%s.lib' % (libname))
666 if os.path.isfile(libfile):
667 fileexists = True
668 break
669 if fileexists:
670 continue
671 # make g77-compiled static libs available to MSVC
672 fileexists = False
673 for libdir in c_library_dirs:
674 libfile = os.path.join(libdir, 'lib%s.a' % (libname))
675 if os.path.isfile(libfile):
676 # copy libname.a file to name.lib so that MSVC linker
677 # can find it
678 libfile2 = os.path.join(self.build_temp, libname + '.lib')
679 copy_file(libfile, libfile2)
680 if self.build_temp not in c_library_dirs:
681 c_library_dirs.append(self.build_temp)
682 fileexists = True
683 break
684 if fileexists:
685 continue
686 log.warn('could not find library %r in directories %s'
687 % (libname, c_library_dirs))

689 # Always use system linker when using MSVC compiler.
690 f_lib_dirs = []
691 for dir in fcompiler.library_dirs:
692 # correct path when compiling in Cygwin but with normal Win
693 # Python
694 if dir.startswith('/usr/lib'):
695 try:
696 dir = subprocess.check_output(['cygpath', '-w', dir])
697 except (OSError, subprocess.CalledProcessError):
698 pass
699 else:
700 dir = filepath_from_subprocess_output(dir)
701 f_lib_dirs.append(dir)

705 for lib in fcompiler.libraries:
706 if not lib.startswith('msvc'):
707 c_libraries.append(lib)
708 p = combine_paths(f_lib_dirs, 'lib' + lib + '.a')
709 if p:
710 dst_name = os.path.join(self.build_temp, lib + '.lib')
711 if not os.path.isfile(dst_name):
712 copy_file(p[0], dst_name)
713 if self.build_temp not in c_library_dirs:
714 c_library_dirs.append(self.build_temp)
