47 @array_function_dispatch(_flip_dispatcher)
48 def fliplr(m):
49 """
50 Reverse the order of elements along axis 1 (left/right).
51  
52 For a 2-D array, this flips the entries in each row in the left/right
53 direction. Columns are preserved, but appear in a different order than
54 before.
55  
56 Parameters
57 ----------
58 m : array_like
59 Input array, must be at least 2-D.
60  
61 Returns
62 -------
63 f : ndarray
64 A view of `m` with the columns reversed. Since a view
65 is returned, this operation is :math:`\\mathcal O(1)`.
66  
67 See Also
68 --------
69 flipud : Flip array in the up/down direction.
70 flip : Flip array in one or more dimensions.
71 rot90 : Rotate array counterclockwise.
72  
73 Notes
74 -----
75 Equivalent to ``m[:,::-1]`` or ``np.flip(m, axis=1)``.
76 Requires the array to be at least 2-D.
77  
78 Examples
79 --------
80 >>> A = np.diag([1.,2.,3.])
81 >>> A
82 array([[1., 0., 0.],
83 [0., 2., 0.],
84 [0., 0., 3.]])
85 >>> np.fliplr(A)
86 array([[0., 0., 1.],
87 [0., 2., 0.],
88 [3., 0., 0.]])
89  
90 >>> A = np.random.randn(2,3,5)
91 >>> np.all(np.fliplr(A) == A[:,::-1,...])
92 True
93  
94 """
95 m = asanyarray(m)
96 if m.ndim < 2:
97 raise ValueError("Input must be >= 2-d.")
98 return m[:, ::-1]

163 def eye(N, M=None, k=0, dtype=float, order='C', *, like=None):
164 """
165 Return a 2-D array with ones on the diagonal and zeros elsewhere.
166  
167 Parameters
168 ----------
169 N : int
170 Number of rows in the output.
171 M : int, optional
172 Number of columns in the output. If None, defaults to `N`.
173 k : int, optional
174 Index of the diagonal: 0 (the default) refers to the main diagonal,
175 a positive value refers to an upper diagonal, and a negative value
176 to a lower diagonal.
177 dtype : data-type, optional
178 Data-type of the returned array.
179 order : {'C', 'F'}, optional
180 Whether the output should be stored in row-major (C-style) or
181 column-major (Fortran-style) order in memory.
182  
183 .. versionadded:: 1.14.0
184 ${ARRAY_FUNCTION_LIKE}
185  
186 .. versionadded:: 1.20.0
187  
188 Returns
189 -------
190 I : ndarray of shape (N,M)
191 An array where all elements are equal to zero, except for the `k`-th
192 diagonal, whose values are equal to one.
193  
194 See Also
195 --------
196 identity : (almost) equivalent function
197 diag : diagonal 2-D array from a 1-D array specified by the user.
198  
199 Examples
200 --------
201 >>> np.eye(2, dtype=int)
202 array([[1, 0],
203 [0, 1]])
204 >>> np.eye(3, k=1)
205 array([[0., 1., 0.],
206 [0., 0., 1.],
207 [0., 0., 0.]])
208  
209 """
210 if like is not None:
211 return _eye_with_like(N, M=M, k=k, dtype=dtype, order=order, like=like)
212 if M is None:
213 M = N
214 m = zeros((N, M), dtype=dtype, order=order)
215 if k >= M:
216 return m
217 if k >= 0:
218 i = k
219 else:
220 i = (-k) * M
221 m[:M-k].flat[i::M+1] = 1
222 return m

234 @array_function_dispatch(_diag_dispatcher)
235 def diag(v, k=0):
236 """
237 Extract a diagonal or construct a diagonal array.
238  
239 See the more detailed documentation for ``numpy.diagonal`` if you use this
240 function to extract a diagonal and wish to write to the resulting array;
241 whether it returns a copy or a view depends on what version of numpy you
242 are using.
243  
244 Parameters
245 ----------
246 v : array_like
247 If `v` is a 2-D array, return a copy of its `k`-th diagonal.
248 If `v` is a 1-D array, return a 2-D array with `v` on the `k`-th
249 diagonal.
250 k : int, optional
251 Diagonal in question. The default is 0. Use `k>0` for diagonals
252 above the main diagonal, and `k<0` for diagonals below the main
253 diagonal.
254  
255 Returns
256 -------
257 out : ndarray
258 The extracted diagonal or constructed diagonal array.
259  
260 See Also
261 --------
262 diagonal : Return specified diagonals.
263 diagflat : Create a 2-D array with the flattened input as a diagonal.
264 trace : Sum along diagonals.
265 triu : Upper triangle of an array.
266 tril : Lower triangle of an array.
267  
268 Examples
269 --------
270 >>> x = np.arange(9).reshape((3,3))
271 >>> x
272 array([[0, 1, 2],
273 [3, 4, 5],
274 [6, 7, 8]])
275  
276 >>> np.diag(x)
277 array([0, 4, 8])
278 >>> np.diag(x, k=1)
279 array([1, 5])
280 >>> np.diag(x, k=-1)
281 array([3, 7])
282  
283 >>> np.diag(np.diag(x))
284 array([[0, 0, 0],
285 [0, 4, 0],
286 [0, 0, 8]])
287  
288 """
289 v = asanyarray(v)
290 s = v.shape
291 if len(s) == 1:
292 n = s[0]+abs(k)
293 res = zeros((n, n), v.dtype)
294 if k >= 0:
295 i = k
296 else:
297 i = (-k) * n
298 res[:n-k].flat[i::n+1] = v
299 return res
300 elif len(s) == 2:
301 return diagonal(v, k)
302 else:
303 raise ValueError("Input must be 1- or 2-d.")

306 @array_function_dispatch(_diag_dispatcher)
307 def diagflat(v, k=0):
308 """
309 Create a two-dimensional array with the flattened input as a diagonal.
310  
311 Parameters
312 ----------
313 v : array_like
314 Input data, which is flattened and set as the `k`-th
315 diagonal of the output.
316 k : int, optional
317 Diagonal to set; 0, the default, corresponds to the "main" diagonal,
318 a positive (negative) `k` giving the number of the diagonal above
319 (below) the main.
320  
321 Returns
322 -------
323 out : ndarray
324 The 2-D output array.
325  
326 See Also
327 --------
328 diag : MATLAB work-alike for 1-D and 2-D arrays.
329 diagonal : Return specified diagonals.
330 trace : Sum along diagonals.
331  
332 Examples
333 --------
334 >>> np.diagflat([[1,2], [3,4]])
335 array([[1, 0, 0, 0],
336 [0, 2, 0, 0],
337 [0, 0, 3, 0],
338 [0, 0, 0, 4]])
339  
340 >>> np.diagflat([1,2], 1)
341 array([[0, 1, 0],
342 [0, 0, 2],
343 [0, 0, 0]])
344  
345 """
346 try:
347 wrap = v.__array_wrap__
348 except AttributeError:
349 wrap = None
350 v = asarray(v).ravel()
351 s = len(v)
352 n = s + abs(k)
353 res = zeros((n, n), v.dtype)
354 if (k >= 0):
355 i = arange(0, n-k, dtype=intp)
356 fi = i+k+i*n
357 else:
358 i = arange(0, n+k, dtype=intp)
359 fi = i+(i-k)*n
360 res.flat[fi] = v
361 if not wrap:
362 return res
363 return wrap(res)

372 def tri(N, M=None, k=0, dtype=float, *, like=None):
373 """
374 An array with ones at and below the given diagonal and zeros elsewhere.
375  
376 Parameters
377 ----------
378 N : int
379 Number of rows in the array.
380 M : int, optional
381 Number of columns in the array.
382 By default, `M` is taken equal to `N`.
383 k : int, optional
384 The sub-diagonal at and below which the array is filled.
385 `k` = 0 is the main diagonal, while `k` < 0 is below it,
386 and `k` > 0 is above. The default is 0.
387 dtype : dtype, optional
388 Data type of the returned array. The default is float.
389 ${ARRAY_FUNCTION_LIKE}
390  
391 .. versionadded:: 1.20.0
392  
393 Returns
394 -------
395 tri : ndarray of shape (N, M)
396 Array with its lower triangle filled with ones and zero elsewhere;
397 in other words ``T[i,j] == 1`` for ``j <= i + k``, 0 otherwise.
398  
399 Examples
400 --------
401 >>> np.tri(3, 5, 2, dtype=int)
402 array([[1, 1, 1, 0, 0],
403 [1, 1, 1, 1, 0],
404 [1, 1, 1, 1, 1]])
405  
406 >>> np.tri(3, 5, -1)
407 array([[0., 0., 0., 0., 0.],
408 [1., 0., 0., 0., 0.],
409 [1., 1., 0., 0., 0.]])
410  
411 """
412 if like is not None:
413 return _tri_with_like(N, M=M, k=k, dtype=dtype, like=like)
414  
415 if M is None:
416 M = N
417  
418 m = greater_equal.outer(arange(N, dtype=_min_int(0, N)),
419 arange(-k, M-k, dtype=_min_int(-k, M - k)))
420  
421 # Avoid making a copy if the requested type is already bool
422 m = m.astype(dtype, copy=False)
423  
424 return m

539 # Originally borrowed from John Hunter and matplotlib
540 @array_function_dispatch(_vander_dispatcher)
541 def vander(x, N=None, increasing=False):
542 """
543 Generate a Vandermonde matrix.
544  
545 The columns of the output matrix are powers of the input vector. The
546 order of the powers is determined by the `increasing` boolean argument.
547 Specifically, when `increasing` is False, the `i`-th output column is
548 the input vector raised element-wise to the power of ``N - i - 1``. Such
549 a matrix with a geometric progression in each row is named for Alexandre-
550 Theophile Vandermonde.
551  
552 Parameters
553 ----------
554 x : array_like
555 1-D input array.
556 N : int, optional
557 Number of columns in the output. If `N` is not specified, a square
558 array is returned (``N = len(x)``).
559 increasing : bool, optional
560 Order of the powers of the columns. If True, the powers increase
561 from left to right, if False (the default) they are reversed.
562  
563 .. versionadded:: 1.9.0
564  
565 Returns
566 -------
567 out : ndarray
568 Vandermonde matrix. If `increasing` is False, the first column is
569 ``x^(N-1)``, the second ``x^(N-2)`` and so forth. If `increasing` is
570 True, the columns are ``x^0, x^1, ..., x^(N-1)``.
571  
572 See Also
573 --------
574 polynomial.polynomial.polyvander
575  
576 Examples
577 --------
578 >>> x = np.array([1, 2, 3, 5])
579 >>> N = 3
580 >>> np.vander(x, N)
581 array([[ 1, 1, 1],
582 [ 4, 2, 1],
583 [ 9, 3, 1],
584 [25, 5, 1]])
585  
586 >>> np.column_stack([x**(N-1-i) for i in range(N)])
587 array([[ 1, 1, 1],
588 [ 4, 2, 1],
589 [ 9, 3, 1],
590 [25, 5, 1]])
591  
592 >>> x = np.array([1, 2, 3, 5])
593 >>> np.vander(x)
594 array([[ 1, 1, 1, 1],
595 [ 8, 4, 2, 1],
596 [ 27, 9, 3, 1],
597 [125, 25, 5, 1]])
598 >>> np.vander(x, increasing=True)
599 array([[ 1, 1, 1, 1],
600 [ 1, 2, 4, 8],
601 [ 1, 3, 9, 27],
602 [ 1, 5, 25, 125]])
603  
604 The determinant of a square Vandermonde matrix is the product
605 of the differences between the values of the input vector:
606  
607 >>> np.linalg.det(np.vander(x))
608 48.000000000000043 # may vary
609 >>> (5-3)*(5-2)*(5-1)*(3-2)*(3-1)*(2-1)
610 48
611  
612 """
613 x = asarray(x)
614 if x.ndim != 1:
615 raise ValueError("x must be a one-dimensional array or sequence.")
616 if N is None:
617 N = len(x)
618  
619 v = empty((len(x), N), dtype=promote_types(x.dtype, int))
620 tmp = v[:, ::-1] if not increasing else v
621  
622 if N > 0:
623 tmp[:, 0] = 1
624 if N > 1:
625 tmp[:, 1:] = x[:, None]
626 multiply.accumulate(tmp[:, 1:], out=tmp[:, 1:], axis=1)
627  
628 return v

649 @array_function_dispatch(_histogram2d_dispatcher)
650 def histogram2d(x, y, bins=10, range=None, normed=None, weights=None,
651 density=None):
652 """
653 Compute the bi-dimensional histogram of two data samples.
654  
655 Parameters
656 ----------
657 x : array_like, shape (N,)
658 An array containing the x coordinates of the points to be
659 histogrammed.
660 y : array_like, shape (N,)
661 An array containing the y coordinates of the points to be
662 histogrammed.
663 bins : int or array_like or [int, int] or [array, array], optional
664 The bin specification:
665  
666 * If int, the number of bins for the two dimensions (nx=ny=bins).
667 * If array_like, the bin edges for the two dimensions
668 (x_edges=y_edges=bins).
669 * If [int, int], the number of bins in each dimension
670 (nx, ny = bins).
671 * If [array, array], the bin edges in each dimension
672 (x_edges, y_edges = bins).
673 * A combination [int, array] or [array, int], where int
674 is the number of bins and array is the bin edges.
675  
676 range : array_like, shape(2,2), optional
677 The leftmost and rightmost edges of the bins along each dimension
678 (if not specified explicitly in the `bins` parameters):
679 ``[[xmin, xmax], [ymin, ymax]]``. All values outside of this range
680 will be considered outliers and not tallied in the histogram.
681 density : bool, optional
682 If False, the default, returns the number of samples in each bin.
683 If True, returns the probability *density* function at the bin,
684 ``bin_count / sample_count / bin_area``.
685 normed : bool, optional
686 An alias for the density argument that behaves identically. To avoid
687 confusion with the broken normed argument to `histogram`, `density`
688 should be preferred.
689 weights : array_like, shape(N,), optional
690 An array of values ``w_i`` weighing each sample ``(x_i, y_i)``.
691 Weights are normalized to 1 if `normed` is True. If `normed` is
692 False, the values of the returned histogram are equal to the sum of
693 the weights belonging to the samples falling into each bin.
694  
695 Returns
696 -------
697 H : ndarray, shape(nx, ny)
698 The bi-dimensional histogram of samples `x` and `y`. Values in `x`
699 are histogrammed along the first dimension and values in `y` are
700 histogrammed along the second dimension.
701 xedges : ndarray, shape(nx+1,)
702 The bin edges along the first dimension.
703 yedges : ndarray, shape(ny+1,)
704 The bin edges along the second dimension.
705  
706 See Also
707 --------
708 histogram : 1D histogram
709 histogramdd : Multidimensional histogram
710  
711 Notes
712 -----
713 When `normed` is True, then the returned histogram is the sample
714 density, defined such that the sum over bins of the product
715 ``bin_value * bin_area`` is 1.
716  
717 Please note that the histogram does not follow the Cartesian convention
718 where `x` values are on the abscissa and `y` values on the ordinate
719 axis. Rather, `x` is histogrammed along the first dimension of the
720 array (vertical), and `y` along the second dimension of the array
721 (horizontal). This ensures compatibility with `histogramdd`.
722  
723 Examples
724 --------
725 >>> from matplotlib.image import NonUniformImage
726 >>> import matplotlib.pyplot as plt
727  
728 Construct a 2-D histogram with variable bin width. First define the bin
729 edges:
730  
731 >>> xedges = [0, 1, 3, 5]
732 >>> yedges = [0, 2, 3, 4, 6]
733  
734 Next we create a histogram H with random bin content:
735  
736 >>> x = np.random.normal(2, 1, 100)
737 >>> y = np.random.normal(1, 1, 100)
738 >>> H, xedges, yedges = np.histogram2d(x, y, bins=(xedges, yedges))
739 >>> # Histogram does not follow Cartesian convention (see Notes),
740 >>> # therefore transpose H for visualization purposes.
741 >>> H = H.T
742  
743 :func:`imshow <matplotlib.pyplot.imshow>` can only display square bins:
744  
745 >>> fig = plt.figure(figsize=(7, 3))
746 >>> ax = fig.add_subplot(131, title='imshow: square bins')
747 >>> plt.imshow(H, interpolation='nearest', origin='lower',
748 ... extent=[xedges[0], xedges[-1], yedges[0], yedges[-1]])
749 <matplotlib.image.AxesImage object at 0x...>
750  
751 :func:`pcolormesh <matplotlib.pyplot.pcolormesh>` can display actual edges:
752

824 def mask_indices(n, mask_func, k=0):
825 """
826 Return the indices to access (n, n) arrays, given a masking function.
827  
828 Assume `mask_func` is a function that, for a square array a of size
829 ``(n, n)`` with a possible offset argument `k`, when called as
830 ``mask_func(a, k)`` returns a new array with zeros in certain locations
831 (functions like `triu` or `tril` do precisely this). Then this function
832 returns the indices where the non-zero values would be located.
833  
834 Parameters
835 ----------
836 n : int
837 The returned indices will be valid to access arrays of shape (n, n).
838 mask_func : callable
839 A function whose call signature is similar to that of `triu`, `tril`.
840 That is, ``mask_func(x, k)`` returns a boolean array, shaped like `x`.
841 `k` is an optional argument to the function.
842 k : scalar
843 An optional argument which is passed through to `mask_func`. Functions
844 like `triu`, `tril` take a second argument that is interpreted as an
845 offset.
846  
847 Returns
848 -------
849 indices : tuple of arrays.
850 The `n` arrays of indices corresponding to the locations where
851 ``mask_func(np.ones((n, n)), k)`` is True.
852  
853 See Also
854 --------
855 triu, tril, triu_indices, tril_indices
856  
857 Notes
858 -----
859 .. versionadded:: 1.4.0
860  
861 Examples
862 --------
863 These are the indices that would allow you to access the upper triangular
864 part of any 3x3 array:
865  
866 >>> iu = np.mask_indices(3, np.triu)
867  
868 For example, if `a` is a 3x3 array:
869  
870 >>> a = np.arange(9).reshape(3, 3)
871 >>> a
872 array([[0, 1, 2],
873 [3, 4, 5],
874 [6, 7, 8]])
875 >>> a[iu]
876 array([0, 1, 2, 4, 5, 8])
877  
878 An offset can be passed also to the masking function. This gets us the
879 indices starting on the first diagonal right of the main one:
880  
881 >>> iu1 = np.mask_indices(3, np.triu, 1)
882  
883 with which we now extract only three elements:
884  
885 >>> a[iu1]
886 array([1, 2, 5])
887

889 m = ones((n, n), int)
890 a = mask_func(m, k)
891 return nonzero(a != 0)
