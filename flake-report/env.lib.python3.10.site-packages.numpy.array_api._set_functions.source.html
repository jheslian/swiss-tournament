38 def unique_all(x: Array, /) -> UniqueAllResult:
39 """
40 Array API compatible wrapper for :py:func:`np.unique <numpy.unique>`.
41  
42 See its docstring for more information.
43 """
44 values, indices, inverse_indices, counts = np.unique(
45 x._array,
46 return_counts=True,
47 return_index=True,
48 return_inverse=True,
49 )
50 # np.unique() flattens inverse indices, but they need to share x's shape

52 inverse_indices = inverse_indices.reshape(x.shape)
