214 def _clean_url_path(path: str, is_local_path: bool) -> str:
215 """
216 Clean the path portion of a URL.
217 """
218 if is_local_path:
219 clean_func = _clean_file_url_path
220 else:
221 clean_func = _clean_url_path_part
222  
223 # Split on the reserved characters prior to cleaning so that
224 # revision strings in VCS URLs are properly preserved.
225 parts = _reserved_chars_re.split(path)
226  
227 cleaned_parts = []
228 for to_clean, reserved in pairwise(itertools.chain(parts, [""])):
229 cleaned_parts.append(clean_func(to_clean))
230 # Normalize %xx escapes (e.g. %2f -> %2F)
231 cleaned_parts.append(reserved.upper())
232  
233 return "".join(cleaned_parts)

236 def _clean_link(url: str) -> str:
237 """
238 Make sure a link is fully quoted.
239 For example, if ' ' occurs in the URL, it will be replaced with "%20",
240 and without double-quoting other characters.
241 """
242 # Split the URL into parts according to the general structure
243 # `scheme://netloc/path;parameters?query#fragment`.
244 result = urllib.parse.urlparse(url)
245 # If the netloc is empty, then the URL refers to a local filesystem path.
246 is_local_path = not result.netloc
247 path = _clean_url_path(result.path, is_local_path=is_local_path)
248 return urllib.parse.urlunparse(result._replace(path=path))
